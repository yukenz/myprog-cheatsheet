## jekyll new blog
# Using jekyll with manual bundler
mkdir my-jekyll-website
cd my-jekyll-website
bundle init

#change gem install directory, for resolve permission problem
bundle config set --local path 'vendor/bundle'

# Add Jekyll
bundle add jekyll

# Create A Jekyll Scaffold
# Kita membutuhkan parameter --force karena folder kita tidak kosong - folder tersebut sudah memiliki beberapa berkas Bundler di dalamnya. Kita jalankan instalasi bundel secara terpisah karena Jekyll akan bingung jika Gemfile sudah ada.
bundle exec jekyll new --force --skip-bundle .
bundle install

# Serve the Site
bundle exec jekyll serve


#Versioning Control
# Jika Anda menyimpan situs baru Anda dalam kontrol versi, Anda sebaiknya mengabaikan folder ./vendor/ dan ./.bundle/ karena folder ini berisi informasi khusus pengguna atau platform. Pengguna baru akan dapat menginstal dependensi yang benar berdasarkan Gemfile dan Gemfile.lock, yang keduanya harus dicentang. Anda dapat menggunakan .gitignore ini untuk memulai, jika Anda ingin

# Ignore metadata generated by Jekyll
_site/
.sass-cache/
.jekyll-cache/
.jekyll-metadata

# Ignore folders generated by Bundler
.bundle/
vendor/
